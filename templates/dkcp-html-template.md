# Шаблон описания компонента для вставки кода ДКЦП в страницу


Используй следующий синтаксис:

`{% set values = ['тут ценности'] %}` 
`{% set segment = ['тут сегмент'] %}` 
`{% set conflict = ['тут конфликт'] %}`
`{% set artifact = ['тут артефакт'] %}` 
`{% set artifactAttrs = ['Атрибут 1', 'Атрибут 2', 'Атрибут 3'] %}` 
`{% set arguments = ['Аргумент 1', 'Аргумент 2'] %}`
`{% set priceConditions = ['Материальный вклад клиента', 'Нематериальный вклад'] %}` 
`{% set exchangeConditions = ['Условие обмена', 'Условие обмена 2'] %}` 
`{% set companyAttrs = ['Атрибут компании 1, 'Атрибут компании 2'] %}` 

`{% set draft = true %} {% include 'dkcp.html' %}`
